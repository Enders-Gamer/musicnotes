let order_of_flats =  ['b', 'e', 'a', 'd', 'g', 'c', 'f'];

let order_of_sharps = ['f', 'c', 'g', 'd', 'a', 'e', 'b'];

let alphabet =        ['a', 'b', 'c', 'd', 'e', 'f', 'g'];

let accidentals = ['double-flat', 'flat', 'natural', 'sharp', 'double-sharp']

function printSequence() {

  clearPage();

  drawStaves();

  let e = 0;

  x = xStart;

  for (ii=0; ii<notes_sequence1.length; ii++) {

    if (x+15>canv1.width-side_margin) {

      x=xStart;

      e+=1;

    }

    let noteName = notes_sequence1[ii].name+notes_sequence1[ii].octave;

    let y = top_margin + 2*staff_spacing + y_values[noteName]*staff_spacing + staves_spacing*e;

    drawNoteHead(x, y);

    drawLedgerLines(x, y, e);

    let noteLetter = notes_sequence1[ii].name;

let draw_accidental = true;

    if (key_signature > 0) {

	  for (i=0; i<key_signature); i++) {

		if (order_of_sharps_name.includes(noteLetter, i) == true) {

		  draw_accidental = false;

		} else {

		  draw_accidental = true;

		}

	  }

	} else if (key_signature < 0) {

	  for (i=0; i>key_signature; i--) {

		if (order_of_flats_name.includes(noteLetter, i) == true) {

		  draw_accidental = false;

		} else {

		  draw_accidental = true;

		}

	  }

	}  

		set_draw_accidental();

	if (draw_accidental == true) {

      drawAccidental(x, y, notes_sequence1[ii].accidental);

	}

    x += note_spacing;

  }

}
